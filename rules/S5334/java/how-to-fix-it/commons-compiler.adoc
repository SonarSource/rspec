=== How to fix it in Commons Compiler

[cols="a"]
|===
h| Non-compliant code example
|
[source,java]
----
import org.codehaus.janino.ScriptEvaluator;

@Controller
public class ExampleController
{
    @GetMapping(value = "/eval")
    public void exec(@RequestParam("code") String code) throws IOException, InvocationTargetException {
        ScriptEvaluator se = new ScriptEvaluator();
        se.cook(code); // Noncompliant
        se.evaluate(null);
    }
}
----
h| Compliant solution
|
[source,java]
----
import org.codehaus.janino.ScriptEvaluator;

@Controller
public class ExampleController
{
    @GetMapping(value = "/eval")
    public void exec(@RequestParam("code") String code) throws IOException, InvocationTargetException {
        ArrayList<String> allowedMethods = new ArrayList<String>();
        allowedMethods.add("test1();");
        allowedMethods.add("test2();");
        if (allowedMethods.contains(code)){
          ScriptEvaluator se = new ScriptEvaluator();
          se.cook(code);
          se.evaluate(null);
        }
    }
}
----
|===
