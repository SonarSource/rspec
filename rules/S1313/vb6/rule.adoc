    sockMain.RemotePort = "80"

End Sub

== Compliant Solution

----
Private Sub cmdConnect_Click()
    sockMain.RemoteHost = txtHost.Text
    sockMain.RemotePort = txtPort.Text
    sockMain.Connect
End Sub
