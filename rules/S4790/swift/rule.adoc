== Sensitive Code Example

For CryptoKit or Swift Crypto
[source,swift]
----
import CryptoKit

let data = Data([0x01, 0x02, 0x03])
let digest = Insecure.MD5.hash(data: data) // Sensitive
----

For CommonCrypto
[source,swift]
----
import CommonCrypto

let data = [0x01, 0x02, 0x03]
let buffer = UnsafeMutablePointer<UInt8>.allocate(capacity: Int(CC_MD5_DIGEST_LENGTH))
CC_MD5(data, (CC_LONG)(strlen(data)), buffer) // Sensitive
free(buffer)
----

== Compliant Solution

For CryptoKit or Swift Crypto
[source,swift]
----
import CryptoKit

let data = Data([0x01, 0x02, 0x03])
let digest = SHA256.hash(data: data)
----

For CommonCrypto
[source,swift]
----
import CommonCrypto

let data = [0x01, 0x02, 0x03]
let buffer = UnsafeMutablePointer<UInt8>.allocate(capacity: Int(CC_SHA256_DIGEST_LENGTH))
CC_SHA256(data, (CC_LONG)(strlen(data)), buffer)
free(buffer)
----
