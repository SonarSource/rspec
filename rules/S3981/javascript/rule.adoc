== Noncompliant Code Example

----
if (someSet.size >= 0) {...} // Noncompliant

if (someMap.size < 0) {...} // Noncompliant

const result = someArray.length >= 0;  // Noncompliant
----

== Compliant Solution

----
if (someSet.size > 0) {...}

if (someMap.size == 0) {...}

const result = someArray.length > 0;
----
