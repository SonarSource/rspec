=== Noncompliant code example

[source,text]
----
if (true) {  
  doSomething(); 
}
...
if (false) {  
  doSomethingElse(); 
}

if (2 < 3 ) { ... }  // Noncompliant; always false

int i = 0;
int j = 0;
// ...
j = foo();

if (j > 0 && i > 0) { ... }  // Noncompliant; always false - i never set after initialization

boolean b = true;
//...
if (b || !b) { ... }  // Noncompliant
----
