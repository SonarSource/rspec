int Abs(int n) // Cognitive Complexity = 3, Computational Complexity = O(1) in space and time
{
    if (n == int.MinValue)  // +1 
    { 
        throw new ArgumentException("The absolute value of int.MinValue is outside of int boundaries");
    } 
    else if (n >= 0)        // +1
    {
        return n; 
    }
    else                    // +1
    { 
        return -n;
    }             
}
