== Noncompliant Code Example

----
void foo(int n, int m) {
  switch (n) { 
    case 0:
      switch (m) {  // Noncompliant; nested switch
        // ...
      }
    case 1:
      // ...
    default:
      // ...
  }
}
----

== Compliant Solution

----
void foo(int n, int m) {
  switch (n) {
    case 0:
      bar(m);      
    case 1:
      // ...
    default:
      // ...
  }
}

void bar(int m){
  switch(m) {
    // ...
  }
}
----
