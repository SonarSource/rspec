== Sensitive Code Example

[source,docker]
----
RUN apt install -y aptitude # Sensitive

RUN aptitude install -y build-essential # Sensitive
----

== Compliant Solution

[source,docker]
----
RUN apt install -y --no-install-recommends aptitude

RUN aptitude install -y --without-recommends build-essential
----
