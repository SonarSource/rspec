----

If compare(getNextValue(), getNextValue()) <> 0 Then ' Noncompliant

----

=== Compliant solution

[source,vbnet]
----
If compare(myPoint.x, myPoint.y) <> 0 Then
    '...
End If

Dim v2 As Integer = getNextValue()
If compare(v1, v2) <> 0 Then
    '...
End If
----
