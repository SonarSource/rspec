== Sensitive Code Example

[source,docker]
----
# Sensitive
ADD --chmod=777 src dst
# Sensitive
COPY --chmod=777 src dst
----

== Compliant Code Example

[source,docker]
----
ADD --chmod=754 src dst
COPY --chown=user:user --chmod=744 src dst
----
