using JetBrains.Annotations;

public class Utils
{
    [TerminatesProgram]
    public void TerminatesProgram() { }

    public void TerminatesProgramIsRespected()
    {
        object myObject = null;
        TerminatesProgram();
        myObject.ToString(); // Compliant: unreachable
    }
}
