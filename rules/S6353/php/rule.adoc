== Noncompliant Code Example

----
preg_match("/[0-9]/", $s); // Noncompliant - same as "/\d/"
preg_match("/[^0-9]/", $s); // Noncompliant - same as "/\D/"
preg_match("/[A-Za-z0-9_]/", $s); // Noncompliant - same as "/\w/" 
preg_match("/[\w\W]/", $s); // Noncompliant - same as "/./" 
preg_match("/a{0,}/", $s); // Noncompliant - same as "/a*/" 
----

== Compliant Solution

----
preg_match("/\d/", $s);
preg_match("/\D/", $s);
preg_match("/\w/", $s);
preg_match("/./", $s);
preg_match("/a*/", $s); 
----
