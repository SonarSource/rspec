== Sensitive Code Example

[source,go,diff-id=1,diff-type=noncompliant]
----
func getName(db *sql.Db, id string) (string, error) {
    var name string
    row := db.QueryRow("SELECT name FROM users WHERE id = " + id).Scan(&name) // Sensitive

    if err := row.Scan(&name); err != nil {
        if err == sql.ErrNoRows {
            return name, fmt.Errorf("No name found for id %s", id)
        }
    }

    return name, nil
}
----

== Compliant Solution

[source,go,diff-id=1,diff-type=compliant]
----
func getName(db *sql.Db, id string) (string, error) {
    var name string
    row := db.QueryRow("SELECT name FROM users WHERE id = ?", id).Scan(&name)

    if err := row.Scan(&name); err != nil {
        if err == sql.ErrNoRows {
            return name, fmt.Errorf("No name found for id %s", id)
        }
    }

    return name, nil
}
----
