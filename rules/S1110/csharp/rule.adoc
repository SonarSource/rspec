== Noncompliant Code Example

----
if (a && ((x + y > 0))) // Noncompliant
{
  //...
}

return ((x + 1));  // Noncompliant
----

== Compliant Solution

----
if (a && (x + y > 0))
{
  //...
}

return x + 1;

return (x + 1);
----
