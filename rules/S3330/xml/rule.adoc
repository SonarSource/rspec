== Sensitive Code Example

----
<session-config>
 <cookie-config>
  <http-only>false</http-only> <!-- Sensitive -->
 </cookie-config>
</session-config>

<session-config>
 <cookie-config> <!-- Sensitive: http-only tag is missing defaulting to false -->
 </cookie-config>
</session-config>
----

== Compliant Solution

----
<session-config>
 <cookie-config>
  <http-only>true</http-only> <!-- Compliant -->
 </cookie-config>
</session-config>
----
