== Sensitive Code Example

=== Java Web Applications

[source,xml]
----
<session-config>
 <cookie-config>
  <http-only>false</http-only> <!-- Sensitive -->
 </cookie-config>
</session-config>

<session-config>
 <cookie-config> <!-- Sensitive: http-only tag is missing defaulting to false -->
 </cookie-config>
</session-config>
----

=== ASP.NET

[source,xml]
----
<configuration>
  <system.web> <!-- Sensitive: httpCookies element with httpOnlyCookies="true" is missing -->
    <!-- ... -->
  </system.web>
  <!-- ... -->
</configuration>
----

== Compliant Solution

=== Java Web Applications

[source,xml]
----
<session-config>
 <cookie-config>
  <http-only>true</http-only> <!-- Compliant -->
 </cookie-config>
</session-config>
----

=== ASP.NET

[source,xml]
----
<configuration>
  <system.web>
    <httpCookies httpOnlyCookies="true" /> <!-- Compliant -->
  </system.web>
  <!-- ... -->
</configuration>
----
