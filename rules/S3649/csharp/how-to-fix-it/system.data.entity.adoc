=== How to fix it in System.Data.Entity

include::../../common/fix/code_rationale.adoc[]

[cols="a"]
|===
h| Non-compliant code example
|
[source,csharp]
----
public class ExampleController : Controller
{
    private readonly UserAccountContext context;

    public IActionResult authenticate(string user, string pass)
    {
        var query = "SELECT * FROM users WHERE user = '" + user + "' AND pass = '" + pass + "'";

        var queryResults = context
            .Database
            .ExecuteSqlCommand(query); // Noncompliant

        if (queryResults == 0)
        {
            return Unauthorized();
        }

        return Ok(AUTHENTICATION_SUCCESS);
    }
}
----
h| Compliant solution
|
[source,csharp]
----
public class ExampleController : Controller
{
    private readonly UserAccountContext Context;

    public IActionResult Authenticate(string user, string pass)
    {
        var query = "SELECT * FROM users WHERE user = {0} AND pass = {1}";

        var queryResults = context
            .Database
            .ExecuteSqlCommand(query, user, pass);

        if (queryResults == 0)
        {
            return Unauthorized();
        }

        return Ok(AUTHENTICATION_SUCCESS);
    }
}
----
