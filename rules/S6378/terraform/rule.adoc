== Sensitive Code Example

----
resource "azurerm_api_management" "sensitive_code" { # Sensitive
  name           = "sensitive code without its identity bloc"
  publisher_name = "Sensitive Company"     
}
----

== Compliant Solution

----
resource "azurerm_api_management" "code" {
  name 			 = "code with its identity bloc"
  publisher_name = "Company"

  identity {
	type = "SystemAssigned, UserAssigned"
}
----
