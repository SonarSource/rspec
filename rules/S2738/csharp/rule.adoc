== Noncompliant Code Example

----
string text = "";
try
{
  text = File.ReadAllText(fileName);
} catch {  // Noncompliant
  // ...
}

try
{
  text = File.ReadAllText(fileName);
} catch (Exception exc) {  // Noncompliant
  // ...
}
----

== Compliant Solution

----
string text = "";
try
{
  text = File.ReadAllText(fileName);
} catch (UnauthorizedAccessException exc) {
  // you do not have the required permission
}
----
