public long ComputeDurationInMilliseconds()
{
  long duration = (((hours * 60) + minutes) * 60 + seconds ) * 1000 ;

}

== Compliant Solution

----
public long ComputeDurationInMilliseconds() 
{
  return (((hours * 60) + minutes) * 60 + seconds ) * 1000 ;
}

public void DoSomething() 
{
  throw new ApplicaitonException();
}
