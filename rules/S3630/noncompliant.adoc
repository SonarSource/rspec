== Noncompliant Code Example

[source,text]
----
  class A { public: virtual ~A(){} };
  class B : public A { public: void doSomething(){} };

  void func(A *a, float f) {
    if (B* b = reinterpret_cast<B*>(a)) { // Noncompliant
      b->doSomething();
    }
    int x = *reinterpret_cast<int*>(f); // Noncompliant
  }
----
