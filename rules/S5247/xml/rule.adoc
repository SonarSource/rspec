== Sensitive Code Example

----
<!-- Java Spring defaultHtmlEscape parameter, in web.xml -->

<context-param>
   <param-name>defaultHtmlEscape</param-name>
   <param-value>false</param-value><!-- Sensitive -->
</context-param>
----

== Compliant Solution

----
<!-- Java Spring defaultHtmlEscape parameter, in web.xml -->

<context-param>
   <param-name>defaultHtmlEscape</param-name>
   <param-value>true</param-value><!-- Compliant -->
</context-param>
----
