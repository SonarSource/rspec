----
$targetPid = (int)$_GET["pid"];

----

== Compliant Solution

[source,php]
----
$targetPid = (int)$_GET["pid"];

// Validate the untrusted PID,
// With a pre-approved list or authorization checks
if (isValidPid($targetPid)) {
    posix_kill($targetPid, 9);
}
----
