----
func fun1() (x, y int) {
  a, b := 1, 2
  b, a = a, b
  return a, b
}

func fun2() (x, y int) {  // Noncompliant; duplicates fun1
  a, b := 1, 2
  b, a = a, b
  return a, b
}
----

[source,go,diff-id=1,diff-type=compliant]
----
func fun1() (x, y int) {
  a, b := 1, 2
  b, a = a, b
  return a, b
}

func fun2() (x, y int) {  // Intent is clear
  return fun1()
}
----
