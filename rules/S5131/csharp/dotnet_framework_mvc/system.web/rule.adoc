[ApiController]
public class PeopleApiController : ControllerBase
{
    [HttpGet("people/{id}")]
    public ActionResult<Person> Get(int id)
    {
        var person = db.People.Find(id);
        return person;
    }

    [HttpPost("people/{id}")]
    public IActionResult Create(Person person)
    {
        db.Add(person);
        db.SaveChanges();
        return Accepted();
    }
}
