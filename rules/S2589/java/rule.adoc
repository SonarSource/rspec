=== Noncompliant code example

[source,text]
----
a = true;
if (a) { // Noncompliant
  doSomething();
}

if (b && a) { // Noncompliant; "a" is always "true"
  doSomething();
}

if (c || !a) { // Noncompliant; "!a" is always "false"
  doSomething();
}

void foo(boolean c, boolean b){
  if (c || (!c && b)){ // Noncompliant; c || (!c && b) is equals to c || b
    doSomething();
  }
}
----

=== Compliant solution

[source,text]
----
a = true;
if (foo(a)) {
  doSomething();
}

if (b) {
  doSomething();
}

if (c) {
  doSomething();
}

void foo(boolean c, boolean b){
  if (c || b){
    doSomething();
  }
}
----
