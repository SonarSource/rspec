== Why is this an issue?

=== Noncompliant code example

[source,cpp]
----
// header.h
int add(int, int);

// file.c
#include header.h

void func(int n, int m) { 
  // ...
  add(n,m);  // Noncompliant
----

=== Compliant solution

[source,cpp]
----
// header.h
int add(int, int);

// file.c
#include header.h

void func(int n, int m) { 
  // ...
  int q = add(n,m); 

  if (add(n,m) > 10) { // also compliant
    // ...
  }
----
