=== Noncompliant code example

[source,kotlin]
----
fun doSomething(file: File, lock: Lock) {
    file.delete() // Noncompliant
    // ...
    lock.tryLock() // Noncompliant
}
----

=== Compliant solution

[source,kotlin]
----
fun doSomething(file: File, lock: Lock) {
    if (!file.delete()) {
        // file delete failed; take appropriate action
    }
    if (!lock.tryLock()) {
        // lock failed; take appropriate action
    }
}
----
