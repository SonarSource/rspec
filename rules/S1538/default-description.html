Why is this an issue?

return, break, continue, and throw statements should be followed by a }, case, or default.

function sayHello() {
  if (true) {
    return;
    var b; // Non-Compliant - this will never be executed
  } else {
    var c;
  }

  while (true) {
    break;
    var d; // Non-Compliant - this will never be executed
  }
}
