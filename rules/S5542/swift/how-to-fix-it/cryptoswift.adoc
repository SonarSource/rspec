== How to fix it in CryptoSwift

=== Code examples

==== Noncompliant code example

include::../../common/fix/aes-noncompliant-example.adoc[]

[source,swift,diff-id=1,diff-type=noncompliant]
----
import CryptoSwift

func encrypt(key: [UInt8], iv: [UInt8], plaintext: [UInt8]) -> [UInt8]? {
    do {
        let aes = try AES(key: key, blockMode: CBC(iv: iv), padding: .pkcs7)  // Noncompliant
        let ciphertext = try aes.encrypt(plaintext)
    } catch {
        // ...
    }
}
----

include::../../common/fix/rsa-noncompliant-example.adoc[]

[source,swift]
----
import CryptoSwift

func encryptRsa(key: [UInt8], plaintext: [UInt8]) -> [UInt8]? {
    do {
        let rsa = try RSA(keySize: 2048)
        let encrypted = try rsa.encrypt(plaintext, padding: .pkcs1)  // Noncompliant
    } catch {
        // ...
    }
}
----


==== Compliant solution

include::../../common/fix/aes-compliant-example.adoc[]

[source,swift,diff-id=1,diff-type=compliant]
----
import CryptoSwift

func encrypt(key: [UInt8], iv: [UInt8], plaintext: [UInt8]) -> [UInt8]? {
    do {
        let aes = try AES(key: key, blockMode: GCM(iv: iv), padding: .noPadding)
        let ciphertext = try aes.encrypt(plaintext)
    } catch {
        // ...
    }
}
----

The recommended RSA mode, RSA-OAEP, is not supported by CryptoSwift. It is recommended to use Apple's https://developer.apple.com/documentation/security/seckeycreateencrypteddata(_:_:_:_:)[Security Framework] instead.
