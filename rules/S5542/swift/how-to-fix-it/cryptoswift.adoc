=== Code examples

==== Noncompliant code example

include::../../common/fix/aes-noncompliant-example.adoc[]

[source,swift,diff-id=1,diff-type=noncompliant]
----
import CryptoSwift

func encrypt(key: Data, iv: Data, plaintext: Data) -> Data? {
    do {
        let aes = try AES(key: key.bytes, blockMode: CBC(iv: iv.bytes), padding: .pkcs7)  // Noncompliant
        let ciphertext = try aes.encrypt(plaintext.bytes)
    } catch {
        // ...
    }
}
----

==== Compliant solution

include::../../common/fix/aes-compliant-example.adoc[]

[source,swift,diff-id=1,diff-type=compliant]
----
import CryptoSwift

func encrypt(key: Data, iv: Data, plaintext: Data) -> Data? {
    do {
        let aes = try AES(key: key.bytes, blockMode: GCM(iv: iv.bytes), padding: .noPadding)
        let ciphertext = try aes.encrypt(plaintext.bytes)
    } catch {
        // ...
    }
}
----

import CryptoSwift

func encryptRsa(key: Data, plaintext: Data) -> Data? {

        // ...

}
