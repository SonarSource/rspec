[source,apex]
----
public class MyClient {
    public void SendRequest(){

        req.setEndpoint('http://example.com');  // Sensitive

    }
}
----

== Compliant Solution

[source,apex]
----
public class MyClient {
    public void SendRequest(){
        HttpRequest req = new HttpRequest();
        req.setEndpoint('https://example.com');
        // ...
    }
}
