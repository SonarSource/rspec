== Sensitive Code Example

For https://aws.amazon.com/ecs/[Amazon ECS]:

----
AWSTemplateFormatVersion: 2010-09-09
Resources:
  EcsTask:
    Type: AWS::ECS::TaskDefinition
    Properties: 
      Family: "service"
      Volumes:
        -
          Name: "storage"
          EFSVolumeConfiguration:
            FilesystemId: !Ref FS
            TransitEncryption: "DISABLED"  # Sensitive
----

AWSTemplateFormatVersion: 2010-09-09

      Volumes:
        -
          Name: "storage"
