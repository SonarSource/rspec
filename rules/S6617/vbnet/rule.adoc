Public Function GetEntities(dbContext As MyDbContext, ids As List(Of Integer)) As List(Of MyEntity)
    Return dbContext.
            MyEntities.
            Where(Function(e) ids.Any(Function(i) e.Id = i)). ' Compliant
            ToList()
End Function
