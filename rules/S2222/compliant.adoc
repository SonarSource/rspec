== Compliant Solution

----
class MyClass 
{
  private object obj = new object();

  public void DoSomething() 
  {
    Monitor.Enter(obj); // Noncompliant
    
    if (IsInitialized())
    {
    }

    Monitor.Exit(obj);
  }
}
----
or
----
class MyClass 
{
  private object obj = new object();

  public void DoSomething() 
  {
    lock(obj) // lock() {...} is easier to use than explicit Monitor calls
    {
        if (IsInitialized())
        {
        }
    }
  }
}
----
