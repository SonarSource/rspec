== Noncompliant Code Example

----
file, err := os.Open("accounts.txt") // Noncompliant
if err != nil {
	log.Fatal(err)
}

bs, err := ioutil.ReadFile("accounts.txt") // Noncompliant
if err != nil {
	return
}
----

== Compliant Solution

----
var location string = prop.Read("myApplication.mySpecificFile")
file, err := os.Open(location)
if err != nil {
	log.Fatal(err)
}
----
